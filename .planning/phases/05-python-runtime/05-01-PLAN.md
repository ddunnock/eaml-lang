    **errors.py** -- Error hierarchy:
    ```python
    class EamlError(Exception): ...  # Base
    class EamlConfigError(EamlError): ...  # Missing keys, bad config
    class EamlValidationError(EamlError):
        def __init__(self, model_id: str, attempts: int, last_response: str, errors: list[str]):
            self.model_id = model_id
            self.attempts = attempts
            self.last_response = last_response
            self.errors = errors
            super().__init__(f"Validation failed after {attempts} attempts for model '{model_id}'. Errors: {'; '.join(errors)}")
    class EamlProviderError(EamlError):
        def __init__(self, provider: str, message: str, status_code: int | None = None):
            self.provider = provider
            self.status_code = status_code
            super().__init__(f"Provider '{provider}' error: {message}")
    ```

    **events.py** -- Telemetry event dataclasses:
    ```python
    from dataclasses import dataclass, field
    from datetime import datetime

    @dataclass
    class CallStartEvent:
        provider: str
        model_id: str
        timestamp: datetime = field(default_factory=datetime.now)

    @dataclass
    class CallEndEvent:
        provider: str
        model_id: str
        duration: float
        token_usage: dict[str, int] | None = None
        timestamp: datetime = field(default_factory=datetime.now)

    @dataclass
    class ValidationFailureEvent:
        provider: str
        model_id: str
        attempt: int
        error: str
        timestamp: datetime = field(default_factory=datetime.now)

    @dataclass
    class ToolCallEvent:
        tool_name: str
        agent_name: str
        timestamp: datetime = field(default_factory=datetime.now)
    ```
