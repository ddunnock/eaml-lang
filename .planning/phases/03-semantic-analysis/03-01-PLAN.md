    use eaml_errors::{Diagnostic, DiagnosticCollector};
    use eaml_lexer::Interner;
    use eaml_parser::{Ast, Program, ParseOutput};

    pub struct AnalysisOutput {

        pub diagnostics: Vec<Diagnostic>,
        pub has_fatal: bool,
    }

    pub fn analyze(parse_output: &ParseOutput, source: &str) -> AnalysisOutput {
        let mut diags = DiagnosticCollector::new(20);
        let symbols = resolver::resolve(
            &parse_output.program,
            &parse_output.ast,
            &parse_output.interner,
            source,
            &mut diags,
        );
        let diagnostics = diags.into_diagnostics();
        let has_fatal = diagnostics.iter().any(|d| d.severity == eaml_errors::Severity::Fatal);
        AnalysisOutput { symbols, diagnostics, has_fatal }
    }

    pub fn assert_has_code(output: &AnalysisOutput, code: ErrorCode) {
        assert!(
            output.diagnostics.iter().any(|d| d.code == code),
            "Expected diagnostic code {} but got: {:?}",
            code,
            output.diagnostics.iter().map(|d| d.code.to_string()).collect::<Vec<_>>()
        );
    }

    pub fn assert_no_errors(output: &AnalysisOutput) {
        let errors: Vec<_> = output.diagnostics.iter()
            .filter(|d| d.severity == Severity::Error || d.severity == Severity::Fatal)
            .collect();
        assert!(errors.is_empty(), "Expected no errors but got: {:?}",
            errors.iter().map(|d| format!("{}: {}", d.code, d.message)).collect::<Vec<_>>());
    }

        output.diagnostics.iter()

    **Pass 1: Register all top-level declarations.**
    Iterate `program.declarations` in order. For each `DeclId`:
    - `DeclId::Model(id)` -> get `ast[id].name`, insert as `SymbolKind::Model(id)`
    - `DeclId::Schema(id)` -> get `ast[id].name`, insert as `SymbolKind::Schema(id)`
    - `DeclId::Prompt(id)` -> get `ast[id].name`, insert as `SymbolKind::Prompt(id)`
    - `DeclId::Tool(id)` -> get `ast[id].name`, insert as `SymbolKind::Tool(id)`
    - `DeclId::Agent(id)` -> get `ast[id].name`, insert as `SymbolKind::Agent(id)`
    - `DeclId::Import(id)` -> extract name from `ImportDecl::Eaml { alias, .. }` or `ImportDecl::Python { alias, .. }`. If alias is Some, use alias. For Python imports, the module template string first text span gives the module name. Register as `SymbolKind::Import(id)`.

    - `DeclId::Error(_)` -> skip

    - **Tool return types**: Check return_type resolves.
    - **Let binding values**: For let bindings, build a sequential scope. Before resolving each let's value expression, only preceding lets (and all top-level declarations) are visible.

    **Helper: `suggest_similar()`**
    ```rust
    fn suggest_similar(name: &str, symbols: &SymbolTable, interner: &Interner) -> Option<String> {
        let threshold = 3;
        symbols.iter()
            .map(|(spur, _)| (strsim::levenshtein(name, interner.resolve(spur)), interner.resolve(spur).to_string()))
            .filter(|(dist, _)| *dist > 0 && *dist <= threshold)
            .min_by_key(|(dist, _)| *dist)
            .map(|(_, s)| s)
    }
    ```
