```rust
pub struct PromptDecl {
    pub name: Spur,
    pub params: Vec<Param>,
    pub requires: Option<RequiresClause>,
    pub return_type: TypeExprId,
    pub body: PromptBody,
    pub span: Span,
}
pub struct PromptBody { pub fields: Vec<PromptField>, pub span: Span }
pub enum PromptField {
    User(TemplateString),
    System(TemplateString),
    Temperature(Span),
    MaxTokens(Span),
    MaxRetries(Span),
}

pub struct ToolDecl {
    pub name: Spur,
    pub params: Vec<Param>,
    pub return_type: TypeExprId,
    pub body: ToolBody,
    pub span: Span,
}
pub enum ToolBody {
    PythonBridge { description: Option<TemplateString>, code_span: Span, span: Span },
    Native { stmts: Vec<ExprId>, span: Span },
    Empty(Span),
}

pub struct AgentDecl { pub name: Spur, pub fields: Vec<AgentField>, pub span: Span }
pub enum AgentField {
    Model(Spur, Span),
    Tools(Vec<(Spur, Span)>, Span),
    System(TemplateString),
    MaxTurns(Span),
    OnError(ErrorPolicy, Span),
}
pub enum ErrorPolicy { Fail, RetryThenFail { retries_span: Span } }

pub struct TemplateString { pub parts: Vec<TemplatePart>, pub span: Span }
pub enum TemplatePart { Text(Span), Interpolation(ExprId, Span) }
pub struct Param { pub name: Spur, pub type_expr: TypeExprId, pub default: Option<ExprId>, pub span: Span }
```

2. Implement `pub fn emit_tool(tool: &ToolDecl, ast: &Ast, interner: &Interner, type_annotations: &TypeAnnotations, source: &str, writer: &mut CodeWriter, imports: &mut ImportTracker)`:
   Per PYB-GEN-01 through PYB-GEN-05:
   - Tool function name: `to_snake_case(interner.resolve(&tool.name))` per CONTEXT.md locked decision
   - Tool functions are sync `def` (not async) per CONTEXT.md locked decision
   - Parameters: emit each param as `name: type_annotation` (keep EAML param names as-is per CONTEXT.md)

   For PythonBridge body:
   (a) Bridge function:
     - `def {snake_name}({params}) -> dict:` (return type is `dict` for bridge functions per PYB-GEN-01)
     - If description present, emit triple-quote docstring per CONTEXT.md locked decision
     - Extract bridge code from `source[code_span.clone()]`
     - Dedent and re-indent to function body level (1 indent = 4 spaces)
     - Emit each line of dedented code with writer

   (b) Wrapper function (per PYB-GEN-01):
     - `def _eaml_call_{snake_name}({params}) -> {ReturnType}:`
     - Body: `result = {snake_name}({param_names})`
     - If return type is a schema: `return {ReturnType}.model_validate(result)`
     - If return type is primitive: type check with isinstance
     - `imports.need_tool_metadata()`

   (c) ToolMetadata (per PYB-GEN-03):
     - `_tool_{snake_name} = ToolMetadata(`
     - `name="{original_eaml_name}"` (camelCase preserved per PYB-GEN-03)
     - `description="{description}"` (from description field, or tool name if absent)
     - `parameters=[{"name": "url", "type": "string"}, ...]` (EAML type names, not Python)
     - `return_type="{eaml_return_type}"`
     - `function=_eaml_call_{snake_name}`

   For Native body: skip (native tool bodies are post-MVP per SEM040 enforcement)
   For Empty body: skip (already caught by SEM040)

3. Implement `pub fn emit_agent(agent: &AgentDecl, ast: &Ast, interner: &Interner, source: &str, writer: &mut CodeWriter, imports: &mut ImportTracker)`:

   - `writer.indent()`
   - Iterate agent.fields:
     - `AgentField::Model(spur, _)` -> `writer.writeln(&format!("model = {}", to_config_name(interner.resolve(spur))))`
     - `AgentField::Tools(tools, _)` -> `writer.writeln(&format!("tools = [{}]", tools.iter().map(|(s,_)| to_snake_case(interner.resolve(s))).collect::<Vec<_>>().join(", ")))` -- reference by direct Python name per CONTEXT.md

     - `AgentField::OnError(ErrorPolicy::Fail, _)` -> `writer.writeln("on_error = \"fail\"")`
     - `AgentField::OnError(ErrorPolicy::RetryThenFail { retries_span }, _)` -> emit both `on_error = "retry_then_fail"` and `on_error_retries = {value}`
   - If no fields, `writer.writeln("pass")`
   - `writer.dedent()`
