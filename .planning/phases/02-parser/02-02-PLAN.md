```rust
fn prefix_bp(kind: TokenKind) -> Option<u8> {
    match kind {
        TokenKind::Bang | TokenKind::Minus => Some(70),
        TokenKind::KwAwait => Some(65),
        _ => None,
    }
}

fn infix_bp(kind: TokenKind) -> Option<(u8, u8)> {
    match kind {
        TokenKind::PipePipe => Some((10, 11)),              // || left-assoc
        TokenKind::AmpAmp => Some((20, 21)),                // && left-assoc
        TokenKind::EqEq | TokenKind::BangEq => Some((30, 31)),    // == != left-assoc per CONTEXT.md
        TokenKind::LAngle | TokenKind::RAngle
        | TokenKind::LessEq | TokenKind::GreaterEq => Some((35, 36)),  // comparisons left-assoc per CONTEXT.md
        TokenKind::Plus | TokenKind::Minus => Some((40, 41)),       // + - left-assoc
        TokenKind::Star | TokenKind::Slash => Some((50, 51)),       // * / left-assoc
        _ => None,
    }
}

fn postfix_bp(kind: TokenKind) -> Option<u8> {
    match kind {
        TokenKind::Dot | TokenKind::LParen | TokenKind::LBracket => Some(80),
        _ => None,
    }
}
```

Note: Comparisons use left-associative BPs (30,31 and 35,36) per CONTEXT.md decision -- semantic analysis (Phase 3) will reject chained comparisons with SEM060.

**`parse_expr(&mut self, min_bp: u8) -> ExprId`** -- Pratt parser entry point:
1. Parse prefix: check `prefix_bp(self.peek())`. If Some(bp), advance, parse_expr(bp) recursively, wrap in UnaryOp or Await.
2. Otherwise parse primary: parse_primary_expr()
3. Loop: check postfix then infix
   - If `postfix_bp(self.peek())` >= min_bp: parse postfix (Dot -> FieldAccess, LParen -> FnCall with argList, LBracket -> Index)
   - Else if `infix_bp(self.peek())` and left_bp >= min_bp: advance op, parse_expr(right_bp), wrap in BinaryOp
   - Else: break
4. Handle PipelineOp (TokenKind::PipelineOp) -- emit SYN081 "Pipeline operator >> is not supported in EAML v0.1." and skip it
5. Handle Pipe (TokenKind::Pipe) in expression context -- emit SYN043 (really Syn044 per error codes) and skip
6. Return ExprId

**`parse_primary_expr(&mut self) -> ExprId`** -- Grammar production [64]:
- IntLit -> Expr::IntLit(span)
- FloatLit -> Expr::FloatLit(span)
- KwTrue -> Expr::BoolLit(true, span)
- KwFalse -> Expr::BoolLit(false, span)
- KwNull -> Expr::NullLit(span)
- Ident(spur) -> Expr::Ident(spur, span)
- LParen -> consume, parse_expr(0), expect RParen, Expr::Paren
- TmplStart -> parse_template_string() -> Expr::TemplateStr(ts)

**`parse_template_string(&mut self) -> TemplateString`** -- Grammar productions [51]-[53]:
1. `let start = self.peek_span().start;`
2. Expect TmplStart (the opening `"`)
3. Loop on peek():
   - TmplText -> advance, push TemplatePart::Text(span)
   - TmplInterpStart -> advance (the `{`), call `self.parse_expr(0)` for full expression, expect TmplInterpEnd (the `}`), push TemplatePart::Interpolation(expr_id, span)
   - TmplEnd -> advance (closing `"`), break
   - Eof -> break (unterminated -- lexer already emitted SYN002)
   - Other -> emit error, break
4. Return TemplateString { parts, span: start..previous_span().end }
